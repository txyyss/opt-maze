\begin{frame}{The Path-Synthesis Problem}
  \begin{columns}[T,totalwidth=\textwidth]
      \begin{column}{0.47\textwidth}
      \setlength{\parindent}{0pt}
      Rasterization gives a finite target region and two prescribed endpoints.\par
      \vspace{0.7em}
      \[
        P\subset\mathbb{N}^{2},
        \qquad
        s,t\in P
      \]
      \[
        s=a_0,a_1,\ldots,a_n=t,
        \qquad a_i\in P,
      \]
      \[
        \lVert a_{i+1}-a_i\rVert_1=1.
      \]
      \par
      \vspace{0.7em}
      \noindent Among all valid paths, maximize \(n\); in the planar case, full coverage gives \(n+1=|P|\).
    \end{column}
    \begin{column}{0.49\textwidth}
      \callout{\linewidth}{\slideaccent{Planar}\\Each grid position is visited at most once.}
      \vspace{0.5em}
      \callout{\linewidth}{\slideaccent{Woven}\\A crossing permits horizontal and vertical visits at one position.}
      \vspace{0.5em}
      \callout{\linewidth}{\slideaccent{Coverage}\\Occupy as many target positions as the path constraints allow.}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Tile-Based Path Representation}
  \centering
  \resizebox{0.9\linewidth}{!}{%
    \input{figures/tiles.tex}%
  }
  \vspace{1em}

  \begin{columns}[T,totalwidth=0.96\textwidth]
    \begin{column}{0.45\textwidth}
      \begin{itemize}
        \item For each \((r,c)\in P\), introduce a tile variable
          \(t_{r,c}\in\{1,\ldots,8\}\).
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{itemize}
        \item Planar paths are the special case where crossing tiles are forbidden.
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.25em}
  \[
    \begin{aligned}
      \mathrm{inRange}(t)&=1\le t\le 8,
      &
      \mathrm{nonWhite}(t)&=\mathbf{1}_{t\ne 8},
      \\
      \mathrm{hasUp}(t)&=t\in\{1,4,5,7\},
      &
      \mathrm{hasDown}(t)&=t\in\{2,3,5,7\},
      \\
      \mathrm{hasRight}(t)&=t\in\{1,2,6,7\},
      &
      \mathrm{hasLeft}(t)&=t\in\{3,4,6,7\}.
    \end{aligned}
  \]
  \vspace{0.05em}
  \[
    C=\sum_{(r,c)\in P}\mathrm{nonWhite}(t_{r,c}),
    \qquad
    C\ge K \quad\text{or maximize } C.
  \]
\end{frame}

\begin{frame}{Local Rules Build a Legal Grid Path}
  \framesubtitle{Boundary constraints}
  \begin{columns}[c,totalwidth=\textwidth]
    \begin{column}{0.32\textwidth}
      \centering
      \input{figures/local-horizontal-boundaries.tex}
    \end{column}
    \begin{column}{0.64\textwidth}
      \begin{itemize}
        \item For each non-endpoint position
          \((r,c)\in P\setminus\{s,t\}\), a missing neighbor disables
          the corresponding exit:
          \[
            \begin{aligned}
              (r,c-1)\notin P
              &\ \Longrightarrow
              \ \neg\,\mathrm{hasLeft}(t_{r,c}),
              \\[0.25em]
              (r,c+1)\notin P
              &\ \Longrightarrow
              \ \neg\,\mathrm{hasRight}(t_{r,c}),
              \\[0.25em]
              (r-1,c)\notin P
              &\ \Longrightarrow
              \ \neg\,\mathrm{hasUp}(t_{r,c}),
              \\[0.25em]
              (r+1,c)\notin P
              &\ \Longrightarrow
              \ \neg\,\mathrm{hasDown}(t_{r,c}).
            \end{aligned}
          \]
        \item The prescribed endpoints \(s\) and \(t\) are the only
          exceptions to these boundary constraints.
        \item At each exception, the boundary side fixes a straight tile:
          horizontal for left/right, and vertical for top/bottom.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Local Constraints Are Not Enough}
  \begin{columns}[c,totalwidth=\textwidth]
    \begin{column}{0.50\textwidth}
      \centering
      \includegraphics[width=0.84\linewidth]{figures/abnormal1.pdf}
    \end{column}
    \begin{column}{0.47\textwidth}
      Local compatibility does not imply global connectivity.
      \vspace{0.9em}
      \begin{itemize}
        \item The endpoint-to-endpoint route satisfies every adjacency
          and boundary rule.
        \item So does the detached loop: no local constraint detects
          that it is disconnected.
      \end{itemize}
      \vspace{0.8em}
      We need a global connectivity certificate.
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Global Connectivity: Parent and Rank}
  \begin{columns}[c,totalwidth=\textwidth]
    \begin{column}{0.29\textwidth}
      \centering
      \resizebox{!}{4.95cm}{%
        \input{figures/parent-rank-chain.tex}%
      }

      \[
        \rho_v >
        \rho_{\operatorname{parent}(v)} >
        \cdots >
        \rho_s=0
      \]
    \end{column}
    \begin{column}{0.68\textwidth}
      For each occupied position \((r,c)\), introduce
      \[
        \rho_{r,c}\in\mathbb Z,
        \qquad
        p^d_{r,c}\in\mathsf{Bool},
        \quad
        d\in\{\kind{R},\kind{L},\kind{U},\kind{D}\}.
      \]
      Here \(\rho_{r,c}\) is its rank; \(p^d_{r,c}\) records whether
      direction \(d\) is selected as a predecessor.

      \begin{itemize}
        \item Root \(s\): \(\rho_s=0\), and no predecessor is selected.
        \item Every other occupied position chooses at least one
          predecessor and has rank in \(\{1,\ldots,|P|\}\):
          \[
            \mathrm{nonWhite}(t_{r,c})=1\wedge(r,c)\ne s
            \Longrightarrow
            \left(\bigvee_d p^d_{r,c}\right)
            \wedge 1\le\rho_{r,c}\le |P|.
          \]
      \end{itemize}

      A predecessor must follow an existing local path edge and have
      smaller rank. For example,
      \[
        p^{\kind{R}}_{r,c}\Longrightarrow
        \mathrm{hasRight}(t_{r,c})
        \wedge\mathrm{hasLeft}(t_{r,c+1})
        \wedge\rho_{r,c+1}<\rho_{r,c}.
      \]
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Crossing Tiles as Virtual Path Elements}
  \begin{columns}[c,totalwidth=\textwidth]
    \begin{column}{0.27\textwidth}
      \centering
      \input{figures/crossing-virtual-elements.tex}
    \end{column}
    \begin{column}{0.70\textwidth}
      A tile activates one or two path elements:
      \[
        \begin{aligned}
          A^{\kind{base}}_{r,c}
          &=\bigl(\mathrm{nonWhite}(t_{r,c})=1\bigr)
            \wedge(t_{r,c}\ne 7),\\
          A^{\kind{h}}_{r,c}=A^{\kind{v}}_{r,c}
          &=(t_{r,c}=7).
        \end{aligned}
      \]

      Each \(k\in\{\kind{base},\kind{h},\kind{v}\}\) gets its own
      parent-rank state:
      \[
        \begin{gathered}
          \rho^k_{r,c}\in\mathbb Z,
          \qquad
          p^{d,k}_{r,c}\in\mathsf{Bool},\\
          A^k_{r,c}\wedge(r,c)\ne s
          \Longrightarrow
          \left(\bigvee_d p^{d,k}_{r,c}\right)
          \wedge 1\le\rho^k_{r,c}\le |P|.
        \end{gathered}
      \]

      \begin{itemize}
        \item The \(\kind{h}\) and \(\kind{v}\) elements permit only
          horizontal and vertical parent directions, respectively.
        \item Coverage remains based on \(\mathrm{nonWhite}(t_{r,c})\),
          so a crossing is counted once.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{SMT-LIB Realism}
  \framesubtitle{A directional parent constraint and its concrete encoding}
  \[
    p^{\kind{R},\kind{h}}_{r,c}\;\Rightarrow\;\bigl(A^{\kind{h}}_{r,c}\wedge \mathrm{hasRight}(t_{r,c})\bigr)\wedge\mathrm{hasLeft}(t_{r,c+1})\wedge\Bigl(A^{\kind{base}}_{r,c+1}\wedge \rho^{\kind{base}}_{r,c+1}<\rho^{\kind{h}}_{r,c}\;\vee\;A^{\kind{h}}_{r,c+1}\wedge \rho^{\kind{h}}_{r,c+1}<\rho^{\kind{h}}_{r,c}\Bigr).
  \]

  In the SMT encoding, this implication expands into a disjunction over
  the possible kinds of the neighboring tile; for instance, with
  \((r,c)=(31,13)\):
\begin{center}
\begin{minipage}{0.75\textwidth}
\begin{lstlisting}
(assert (=> pR_31_13_h
  (or (and (= tile_31_14 7)
           (= tile_31_13 7) (hasRight tile_31_13) (hasLeft tile_31_14)
           (< rank_31_14_h rank_31_13_h))
      (and (and (not (= tile_31_14 7)) (= (nonWhite tile_31_14) 1))
           (= tile_31_13 7) (hasRight tile_31_13) (hasLeft tile_31_14)
           (< rank_31_14_base rank_31_13_h)))))
\end{lstlisting}
\end{minipage}
\end{center}
  \begin{itemize}
    \item The constraints are emitted as SMT-LIB, not pseudocode.
    \item This is the horizontal parent implication for a crossing virtual element.
    \item The neighbor can be another crossing \(\kind{h}\) element or an ordinary \(\kind{base}\) element.
  \end{itemize}
\end{frame}

\begin{frame}{Maze Completion from a Prescribed Solution Path}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.43\textwidth}
      \centering
      \includegraphics[height=0.75\paperheight,keepaspectratio]{figures/crossed-s.pdf}
    \end{column}
    \begin{column}{0.54\textwidth}
      \begin{itemize}
        \item Read the SMT model: each value \(t_{r,c}\) selects a tile, and the tile assignment draws the prescribed solution path.
        \item Planar: insert the solution-path edges first, then attach each remaining grid position by a Wilson-style loop-erased random walk that stops at the existing tree.
        \item Woven: insert required crossings and solution links first, then use Kruskal-style completion on the expanded over/under graph.
        \item The completed passage graph is a tree, so the entry--exit solution is unique.
        \item Render the passage graph using standard walls and passages, with inset corridors used to depict woven crossings.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
