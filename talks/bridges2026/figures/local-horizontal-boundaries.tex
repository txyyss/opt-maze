  },

    font=\normalsize
  },

\def\half{0.625}

\newcommand{\leftboundarytile}[3]{%
  \begin{scope}[shift={(-1.35,#1)}]
    \draw[tile] (-\half,-\half) rectangle (\half,\half);
    #3
    \node[tilenum,anchor=west] at (0.88,0) {#2};
  \end{scope}
}

\newcommand{\rightboundarytile}[3]{%
  \begin{scope}[shift={(1.35,#1)}]
    \draw[tile] (-\half,-\half) rectangle (\half,\half);
    #3
    \node[tilenum,anchor=east] at (-0.88,0) {#2};
  \end{scope}
}
