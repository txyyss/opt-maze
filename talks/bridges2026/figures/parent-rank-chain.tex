\begin{tikzpicture}[
  x=1.6cm,
  y=1.6cm,
  grid line/.style={draw=black,line width=0.5pt},
  pathseg/.style={draw=black,line width=6pt,line cap=butt,line join=miter},
  parent/.style={-{Latex[length=2.8mm,width=1.8mm]},draw=black,line width=1.2pt},
  rank/.style={text=black!28,font=\fontsize{40}{40}\selectfont},
  note/.style={font=\fontsize{22}{22}\selectfont}
]
  \path[use as bounding box] (-0.05,-0.45) rectangle (3.05,5.05);

  % Draw the complete 3-by-5 tile grid explicitly.
  \draw[grid line] (0,0) rectangle (3,5);
  \foreach \x in {1,2} {
    \draw[grid line] (\x,0) -- (\x,5);
  }
  \foreach \y in {1,2,3,4} {
    \draw[grid line] (0,\y) -- (3,\y);
  }

  % Ranks sit below the local path fragments.
  \foreach \x/\y/\r in {
    0.5/0.5/0,
    1.5/0.5/1,
    1.5/1.5/2,
    1.5/2.5/3,
    1.5/3.5/4,
    1.5/4.5/5,
    2.5/4.5/6
  } {
    \node[rank] at (\x,\y) {\(\r\)};
  }

  % Enter at the lower left, turn upward in the middle column,
  % then turn right across the top row.
  \begin{scope}[shift={(0,0)}]
    \draw[pathseg] (0,0.5) -- (1,0.5);
  \end{scope}
  \begin{scope}[shift={(1,0)}]
    \draw[pathseg] (0,0.5) -- (0.5,0.5) -- (0.5,1);
  \end{scope}

  \begin{scope}[shift={(1,1)}]
    \draw[pathseg] (0.5,0) -- (0.5,1);
  \end{scope}
  \begin{scope}[shift={(1,2)}]
    \draw[pathseg] (0.5,0) -- (0.5,1);
  \end{scope}
  \begin{scope}[shift={(1,3)}]
    \draw[pathseg] (0.5,0) -- (0.5,1);
  \end{scope}
  \begin{scope}[shift={(1,4)}]
    \draw[pathseg] (0.5,0) -- (0.5,0.5) -- (1,0.5);
  \end{scope}
  \begin{scope}[shift={(2,4)}]
    \draw[pathseg] (0,0.5) -- (1,0.5);
  \end{scope}

  \node[note] at (0.5,-0.24) {\(s\)};

  % Parent arrows occupy the unused half of each tile.
  \draw[parent] (1.80,0.22) -- (1.20,0.22);
  \draw[parent] (1.78,1.80) -- (1.78,1.20);
  \draw[parent] (1.22,2.80) -- (1.22,2.20);
  \draw[parent] (1.78,3.80) -- (1.78,3.20);
  \draw[parent] (1.22,4.80) -- (1.22,4.20);
  \draw[parent] (2.80,4.22) -- (2.20,4.22);
\end{tikzpicture}
